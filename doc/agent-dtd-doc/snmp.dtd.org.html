<!--
     Configuration for the Ultra Messaging SNMP Agent Daemon.
     @title Ultra Messaging SNMP Agent Daemon
     @root tnwmonsnmp
-->
<!-- The root element of the Ultra Messaging SNMP agent configuration. -->
<!ELEMENT tnwmonsnmp (daemon?, monitor?, (master-agent|subagent))>
<!-- @attr version Ultra Messaging SNMP Agent DTD version used. -->
<!ATTLIST tnwmonsnmp
          version (1.0) #REQUIRED
>
<!-- Specifies the global daemon configuration. -->
<!ELEMENT daemon (log?, uid?, gid?, pidfile?, lbm-license-file?)>
<!-- Indicates the file where log messages are sent. -->
<!ELEMENT log ( #PCDATA )>
<!--
      @attr type Type of logging. For syslog and console, the filename is ignored.
      @attr xml:space How whitespace is to be handled. "default" trims leading
            and trailing whitespace, and compress multiple whitespace characters
            into a single character. "preserve" preserves the whitespace exactly
            as read.
-->
<!ATTLIST log
          type (file | syslog | console) "console"
          xml:space (default | preserve) "default"
>
<!-- Pathname for daemon process ID (PID) file. -->
<!ELEMENT pidfile ( #PCDATA )>
<!--
      @attr xml:space How whitespace is to be handled. "default" trims leading
            and trailing whitespace, and compress multiple whitespace characters
            into a single character. "preserve" preserves the whitespace exactly
            as read.
-->
<!ATTLIST pidfile xml:space (default | preserve) "default">
<!-- User ID (UID) for daemon process (if run as root) -->
<!ELEMENT uid ( #PCDATA )>
<!-- Group ID (GID) for daemon process (if run as root) -->
<!ELEMENT gid ( #PCDATA )>
<!-- Specifies the LBM license file. -->
<!ELEMENT lbm-license-file ( #PCDATA )>
<!--
      @attr xml:space How whitespace is to be handled. "default" trims leading
            and trailing whitespace, and compress multiple whitespace characters
            into a single character. "preserve" preserves the whitespace exactly
            as read.
-->
<!ATTLIST lbm-license-file xml:space (default | preserve) "default">
<!-- Monitoring configuration. -->
<!ELEMENT monitor (lbm-config-file?)>
<!--
      @attr max-age Maximum age (in seconds) of a statistics entry with no updates before it is purged.
      @attr purge-interval Interval (in seconds) between purge checks.
      @attr topic Topic on which statistics are delivered.
-->
<!ATTLIST monitor
          max-age CDATA "60"
          purge-interval CDATA "15"
          topic CDATA "/29west/statistics"
>
<!-- UM configuration file to be used for monitoring. -->
<!ELEMENT lbm-config-file ( #PCDATA )>
<!--
      @attr xml:space How whitespace is to be handled. "default" trims leading
            and trailing whitespace, and compress multiple whitespace characters
            into a single character. "preserve" preserves the whitespace exactly
            as read.
-->
<!ATTLIST lbm-config-file xml:space (default | preserve) "default">
<!-- SNMP master agent configuration. -->
<!ELEMENT master-agent (interface?)>
<!--
      @attr community SNMP V2c RO community name
      @attr port UDP port on which to receive SNMP queries
-->
<!ATTLIST master-agent
          community CDATA "public"
          port CDATA "11161"
>
<!-- Interface to which to bind the listening SNMP port. -->
<!ELEMENT interface ( #PCDATA )>
<!-- SNMP subagent configuration. -->
<!ELEMENT subagent EMPTY>
<!-- @attr agentx-socket Socket used to communicate with master agent.
           For Windows, this defaults to tcp:localhost:705, and
           for Unix/Linux this defaults to unix:/var/agentx/master.-->
<!ATTLIST subagent agentx-socket CDATA "unix:/var/agentx/master">
